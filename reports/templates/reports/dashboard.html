let mediaRecorder; let audioChunks = []; const recordBtn = document.getElementById('recordBtn'); const statusLabel = document.getElementById('statusLabel'); const audioInput = document.getElementById('audioInput'); const audioPlayback = document.getElementById('audioPlayback'); recordBtn.addEventListener('click', async () => { if (!mediaRecorder || mediaRecorder.state === 'inactive') { // Début de l'enregistrement const stream = await navigator.mediaDevices.getUserMedia({ audio: true }); mediaRecorder = new MediaRecorder(stream); audioChunks = []; mediaRecorder.ondataavailable = (event) => audioChunks.push(event.data); mediaRecorder.onstop = () => { const audioBlob = new Blob(audioChunks, { type: 'audio/wav' }); // On place le blob dans l'input file virtuel pour Django const file = new File([audioBlob], "temp_audio.wav", { type: "audio/wav" }); const container = new DataTransfer(); container.items.add(file); audioInput.files = container.files; // On permet à l'agent d'écouter audioPlayback.src = URL.createObjectURL(audioBlob); audioPlayback.classList.remove('d-none'); }; mediaRecorder.start(); recordBtn.classList.add('btn-danger', 'recording-active'); recordBtn.classList.remove('btn-outline-danger'); statusLabel.classList.remove('d-none'); } else { // Fin de l'enregistrement mediaRecorder.stop(); recordBtn.classList.remove('btn-danger', 'recording-active'); recordBtn.classList.add('btn-outline-success'); statusLabel.classList.add('d-none'); } });
